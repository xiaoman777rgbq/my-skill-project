📚 JavaScript 变量类型

1. 字符串 String

var name = "你好世界";
var greeting = 'Hello';

2. 数字 Number

var age = 25;
var price = 99.9;

3. 布尔值 Boolean

var isActive = true;
var isEmpty = false;

4. 未定义 Undefined

var notAssigned;
console.log(notAssigned);

5. 空值 Null

var empty = null;
console.log(empty);

6. 数组 Array

var fruits = ["苹果", "香蕉"];

7. 对象 Object

var person = {
  name: "小明",
  age: 20
};

8. typeof 检测类型

typeof "hello"  // "string"
typeof 123      // "number"
typeof true     // "boolean"

💡 类型转换示例

// 字符串转数字
var num = Number("123");     // 123
var nan = Number("abc");      // NaN

// 数字转字符串
var str = String(123);       // "123"
var str2 = (123).toString(); // "123"

// 强制类型转换
var bool = Boolean(1);       // true
var bool2 = Boolean(0);      // false
