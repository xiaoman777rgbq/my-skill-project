📚 JavaScript 函数

1. 函数声明（function keyword）

// 基本语法
function greet(name) {
    return "你好，" + name + "！";
}
console.log(greet("小明"));

// 带默认参数
function sayHello(name = "世界") {
    return "你好，" + name + "！";
}

2. 函数表达式

// 将函数赋值给变量
var add = function(a, b) {
    return a + b;
};
console.log(add(3, 5));  // 8

// 箭头函数（ES6）
var multiply = (a, b) => a * b;
console.log(multiply(4, 5));  // 20

3. 箭头函数（ES6）

// 无参数
var greet = () => "你好！";

// 单参数（可省略括号）
var double = x => x * 2;

// 多参数
var sum = (a, b) => a + b;

// 多行函数体
var calc = (a, b) => {
    var result = a * b;
    return result;
};

4. 函数参数

// arguments 对象（传统方式）
function sumAll() {
    var total = 0;
    for (var i = 0; i < arguments.length; i++) {
        total += arguments[i];
    }
    return total;
}

// rest 参数（ES6）
function sumRest(...nums) {
    return nums.reduce((a, b) => a + b, 0);
}

5. 回调函数

// 回调函数示例
function processData(data, callback) {
    return callback(data);
}

var result = processData(5, function(x) {
    return x * 2;
});
console.log(result);  // 10

// 使用箭头函数的回调
var numbers = [1, 2, 3, 4, 5];
var doubled = numbers.map(x => x * 2);

6. 闭包函数

// 闭包：函数访问外部变量
function createCounter() {
    var count = 0;
    return function() {
        count++;
        return count;
    };
}

var counter = createCounter();
console.log(counter());  // 1
console.log(counter());  // 2
console.log(counter());  // 3

7. 递归函数

// 计算阶乘
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// 计算斐波那契数列
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}
💡 重要概念：
  • 函数声明：会被提升（hoisting），可以在声明前调用
  • 函数表达式：不会被提升，只有赋值后才能调用
  • 箭头函数：没有自己的 this，不能用作构造函数
  • 闭包：让函数记住创建时的作用域

📝 练习题

练习 1：阶乘函数

题目：编写一个函数计算 n 的阶乘（n!）

// 阶乘公式: n! = n × (n-1) × (n-2) × ... × 1
// 例如: 5! = 5 × 4 × 3 × 2 × 1 = 120
💡 答案提示（点击展开）

两种实现方式：

// 方法1：递归（简洁但有栈溢出风险）
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// 方法2：循环（更安全）
function factorial(n) {
    var result = 1;
    for (var i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

练习 2：防抖函数（实际应用）

题目：实现一个简单的防抖函数，延迟执行

// 防抖原理：事件触发后，等待 N 秒再执行
// 如果在等待期间再次触发，重新计时
💡 答案提示（点击展开）

闭包应用场景：搜索框输入、按钮防重复点击

function debounce(func, delay) {
    var timer = null;  // 用闭包保存定时器
    return function() {
        clearTimeout(timer);  // 清除之前的定时器
        timer = setTimeout(function() {
            func();  // 执行函数
        }, delay);
    };
}

// 使用示例
var handleSearch = debounce(function() {
    console.log("执行搜索...");
}, 1000);
