📚 JavaScript 流程控制

1. if 条件语句

var score = 85;
if (score >= 90) {
    console.log("优秀");
} else if (score >= 60) {
    console.log("及格");
} else {
    console.log("不及格");
}

2. switch 多分支语句

var day = 3;
switch (day) {
    case 1: console.log("星期一"); break;
    case 2: console.log("星期二"); break;
    case 3: console.log("星期三"); break;
    default: console.log("其他");
}

3. for 循环

// 输出 1-5 的平方
for (var i = 1; i <= 5; i++) {
    console.log(i + "的平方是 " + (i * i));
}

// 遍历数组
var fruits = ["苹果", "香蕉", "橙子"];
for (var i = 0; i < fruits.length; i++) {
    console.log(fruits[i]);
}

4. while / do-while 循环

// while 循环
var count = 1;
while (count <= 3) {
    console.log("while: " + count);
    count++;
}

// do-while 循环（至少执行一次）
var n = 1;
do {
    console.log("do-while: " + n);
    n++;
} while (n <= 3);

5. for...in 和 for...of 循环

// for...in 遍历对象属性
var person = {name: "小明", age: 20, city: "北京"};
for (var key in person) {
    console.log(key + ": " + person[key]);
}

// for...of 遍历数组元素（ES6）
var colors = ["红", "绿", "蓝"];
for (var color of colors) {
    console.log(color);
}

6. break 和 continue

// break: 跳出整个循环
for (var i = 1; i <= 5; i++) {
    if (i === 3) break;
    console.log("break at " + i);
}

// continue: 跳过当前迭代
for (var j = 1; j <= 5; j++) {
    if (j === 3) continue;
    console.log("continue at " + j);
}
