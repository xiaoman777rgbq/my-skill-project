📚 JavaScript 表达式

1. 算术表达式

var a = 10, b = 3;
console.log(a + b);   // 加法: 13
console.log(a - b);   // 减法: 7
console.log(a * b);   // 乘法: 30
console.log(a / b);   // 除法: 3.333...
console.log(a % b);   // 取余: 1
console.log(a ** b);  // 幂运算: 1000

2. 赋值表达式

var x = 10;
x += 5;    // x = 15
x -= 3;    // x = 12
x *= 2;    // x = 24
x /= 4;    // x = 6
x %= 4;    // x = 2

3. 比较表达式（返回布尔值）

var a = 5, b = '5';
console.log(a == b);   // 相等（值）: true
console.log(a === b);  // 严格相等: false
console.log(a != b);   // 不等: false
console.log(a !== b);  // 严格不等: true
console.log(a > 3);    // 大于: true
console.log(a <= 5);   // 小于等于: true

4. 逻辑表达式

var x = true, y = false;
console.log(x && y);  // 逻辑与: false
console.log(x || y);  // 逻辑或: true
console.log(!x);      // 逻辑非: false

// 短路运算
var name = null;
var result = name || "默认值";  // "默认值"

5. 条件表达式（三元运算符）

var age = 20;
var status = age >= 18 ? "成年人" : "未成年人";
console.log(status);  // "成年人"

6. 字符串表达式

var firstName = "张";
var lastName = "三";
var fullName = firstName + lastName;  // "张三"

var age = 25;
var info = "年龄：" + age;  // "年龄：25"

// 模板字符串（ES6）
var template = `姓名：${fullName}，年龄：${age}`;

7. 自增/自减表达式

var count = 5;
console.log(count++);  // 5 (先返回，后自增)
console.log(count);   // 6

var count2 = 5;
console.log(++count2); // 6 (先自增，后返回)
console.log(--count2); // 5 (自减)
